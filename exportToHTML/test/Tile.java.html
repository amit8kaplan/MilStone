142           * @return randon tile
143           * O(n)
144           */
145          public Tile getRand(){
146              if (CountTileInArray <=0)
147                  return null;
148              Random rand = new Random();
149              int randPosition;
150              if (CountTileInArray >20) {
151                  boolean ifPositionEmpty = true;//meaning true :empty!
152                  while (ifPositionEmpty) {
153                      randPosition = rand.nextInt(26);
154                      if (checkIfPositionInLetterGameIsEmpty(randPosition))
155                      {
156                          ifPositionEmpty = false;
157                          return this.arrayTile[randPosition];
158                      }
159                  }
160              }
161              randPosition = rand.nextInt(26);
162              for (int i=randPosition; i<LettersGame.length; i++)
163                  if (checkIfPositionInLetterGameIsEmpty(i))
164                      return this.arrayTile[i];
165              for (int i=0; i<randPosition; i++)
166                  if (checkIfPositionInLetterGameIsEmpty(i))
167                      return this.arrayTile[i];
168              return null;
169          }
